//오랜지색 pwm / 빨간색 vcc(5v) / 갈색 ground
#include <Servo.h>

Servo myser;
int value = 0;
//+30 왼쪽 30도 / -30 오른쪽 30도
void setup(){
  Serial.begin(9600);
  myser.attach(8);
}

void loop(){
  if(Serial.available()){
    char in_data;
    in_data = Serial.read();
   if(in_data=='1'){ //lock 오른쪽으로 각도
     value -= 90;
   } 
   else if(in_data=='3'){ //unlock 왼쪽으로 각도
     value += 90;
   }
   else value = 0; //각도 초기화

   myser.write(value);
  }
}

void setup() {
  // put your setup code here, to run once:
  pinMode(LED_R1, OUTPUT);
  pinMode(LED_Y1, OUTPUT);
  pinMode(LED_G1, OUTPUT);
  
  pinMode(LED_R2, OUTPUT);
  pinMode(LED_Y2, OUTPUT);
  pinMode(LED_G2, OUTPUT);  
  Serial.begin(9600);
}
 
void loop() {
  // put your main code here, to run repeatedly:
 if(Serial.available()){
  char in_data;
  in_data = Serial.read();

  if(in_data=='0') turnOffAll1();
  else if(in_data=='1') digitalWrite(LED_R1, HIGH);  //잠금 불
  else if(in_data=='2') digitalWrite(LED_Y1, HIGH);
  else if(in_data=='3') digitalWrite(LED_G1, HIGH);

  if(in_data=='4') turnOffAll2();
  else if(in_data=='5') digitalWrite(LED_R2, HIGH);
  else if(in_data=='6') digitalWrite(LED_Y2, HIGH);
  else if(in_data=='7') digitalWrite(LED_G2, HIGH);
  
 }
 
}
 
void turnOffAll1() {
  digitalWrite(LED_R1, LOW);
  digitalWrite(LED_Y1, LOW);
  digitalWrite(LED_G1, LOW);
}
void turnOffAll2() {
  digitalWrite(LED_R2, LOW);
  digitalWrite(LED_Y2, LOW);
  digitalWrite(LED_G2, LOW);
}
